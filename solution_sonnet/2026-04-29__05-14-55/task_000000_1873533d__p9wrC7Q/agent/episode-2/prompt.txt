#!/bin/bash

# Check for correct number of arguments
if [ $# -ne 2 ]; then
    echo "Usage: $0 <file1.env> <file2.env>" >&2
    exit 1
fi

# Check if both files exist
if [ ! -f "$1" ]; then
    echo "Error: File '$1' not found" >&2
    exit 1
fi

if [ ! -f "$2" ]; then
    echo "Error: File '$2' not found" >&2
    exit 1
fi

# Use associative array to store key-value pairs
declare -A env_vars
declare -a keys_order

# Function to parse and store env file
parse_env_file() {
    local file="$1"
    while IFS= read -r line || [ -n "$line" ]; do
        # Skip empty lines and comments
        [[ -z "$line" || "$line" =~ ^[[:space:]]*# ]] && continue

        # Extract key=value (handle values with = in them)
        if [[ "$line" =~ ^([^=]+)=(.*)$ ]]; then
            key="${BASH_REMATCH[1]}"
            value="${BASH_REMATCH[2]}"

            # Only add to keys_order if it's a new key
            if [ -z "${env_vars[$key]+x}" ]; then
                keys_order+=("$key")
            fi

            env_vars["$key"]="$value"
        fi
    done < "$file"
}

# Parse both files (second file overwrites first)
parse_env_file "$1"
parse_env_file "$2"

# Output merged result in order
for key in "${keys_order[@]}"; do
    echo "${key}=${env_vars[$key]}"
done
