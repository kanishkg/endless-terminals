#!/bin/bash

# Backup script that handles permission issues gracefully
SOURCE_DIR="/home/user/projects"
BACKUP_DIR="/home/user/backups"
MANIFEST="${BACKUP_DIR}/manifest.txt"

# Create backup directory if it doesn't exist
mkdir -p "${BACKUP_DIR}"

# Clear or create manifest
> "${MANIFEST}"

echo "Starting backup from ${SOURCE_DIR} to ${BACKUP_DIR}..."

# Find all files, suppressing permission errors
find "${SOURCE_DIR}" -type f 2>/dev/null | while IFS= read -r file; do
    # Check if file is readable
    if [ -r "${file}" ]; then
        # Get relative path
        rel_path="${file#${SOURCE_DIR}/}"
        dest_file="${BACKUP_DIR}/${rel_path}"
        dest_dir=$(dirname "${dest_file}")

        # Create destination directory structure
        mkdir -p "${dest_dir}" 2>/dev/null

        # Copy the file
        if cp "${file}" "${dest_file}" 2>/dev/null; then
            # Get owner:group information
            owner_group=$(stat -c '%U:%G' "${file}" 2>/dev/null)
            if [ -z "${owner_group}" ]; then
                owner_group="unknown:unknown"
            fi

            # Write to manifest
            echo "${rel_path} ${owner_group}" >> "${MANIFEST}"
            echo "Copied: ${rel_path} (${owner_group})"
        fi
    fi
done

echo "Backup complete. Manifest written to ${MANIFEST}"
