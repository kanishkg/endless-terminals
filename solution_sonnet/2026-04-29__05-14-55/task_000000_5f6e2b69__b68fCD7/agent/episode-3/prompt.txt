cd "$(dirname "$0")/artifacts"
failures=0

    if [[ -z "$filename" ]]; then continue; fi
    actual_sum=$(sha256sum "$filename" 2>/dev/null | cut -d' ' -f1)
    if [[ "$actual_sum" != "$expected_sum" ]]; then

    fi
done < ../checksums.manifest

exit $failures
