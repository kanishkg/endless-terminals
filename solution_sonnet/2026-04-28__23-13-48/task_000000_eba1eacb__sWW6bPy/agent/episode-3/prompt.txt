#!/usr/bin/env python3
import json

def main():
    results = {"hosts": {}}

    with open("results/ping.txt") as f:
        for line in f:
            host, status = line.strip().split(": ")
            results["hosts"][host] = {"ping": status}

    with open("results/ports.txt") as f:
        for line in f:
            parts = line.strip().split()
            host_port = parts[0].split(":")
            host = host_port[0]
            port = host_port[1]
            status = parts[1]
            if host not in results["hosts"]:
                results["hosts"][host] = {}
            if "ports" not in results["hosts"][host]:
                results["hosts"][host]["ports"] = {}
            results["hosts"][host]["ports"][port] = status

    with open("results/status.json", "w") as f
        json.dump(results, f, indent=2)

    print("Report generated: results/status.json")

if __name__ == "__main__":
    main()
