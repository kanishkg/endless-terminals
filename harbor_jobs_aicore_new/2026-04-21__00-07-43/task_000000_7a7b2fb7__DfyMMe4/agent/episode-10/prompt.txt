def get_installed_version(pkg):
    """Get installed version of a package"""
    try:
        result = subprocess.run(['dpkg-query', '-W', '-f=${Version}', pkg],
                               capture_output=True, text=True, timeout=10)
        if result.returncode == 0:
            return result.stdout.strip()
    except:
        pass
    return "unknown"

def has_security_update(pkg):
    """Check if package has a security update available"""
    try:
        result = subprocess.run(['apt-cache', 'policy', pkg],
                               capture_output=True, text=True, timeout=10)
        output = result.stdout
        # Check if there's a security pocket version available
        if '-security' in output:
            lines = output.split('\n')
            installed_ver = None
            candidate_ver = None
            for line in lines:
                if 'Installed:' in line:
                    installed_ver = line.split(':')[1].strip()
                elif 'Candidate:' in line:
                    candidate_ver = line.split(':')[1].strip()
            if installed_ver and candidate_ver and installed_ver != candidate_ver:
                # Check if candidate comes from security
                for line in lines:
                    if candidate_ver in line and '-security' in line:
                        return True
        return False
    except:
        return False
