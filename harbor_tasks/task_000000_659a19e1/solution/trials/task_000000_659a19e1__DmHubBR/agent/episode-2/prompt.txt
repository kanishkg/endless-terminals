# Get current timestamp in YYYY-MM-DD HH:MM:SS format
TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')

USAGE_PERCENT=$(df / | awk 'NR==2 {gsub(/%/,""); print $5}')
